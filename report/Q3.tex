\renewcommand{\parttitle}{Wing deformed shape -- Strut effect}
\section*{\parttitle}

Compare wing deformation for $\beta_0 = 10^\circ$ at $q = 0.4\,q_D^{(0)}$, where $q_D^{(0)}$ is the divergence dynamic pressure without strut. 

\paragraph{Displacement field.}
\begin{equation}
z(x,y) = w(y) - x\,\theta(y) = \left(\frac{y}{b}\right)^2 \phi_w - x\,\frac{y}{b}\,\phi_\theta
\end{equation}

\paragraph{Aeroelastic equilibrium.}
Inverting $\mathbf{K}_{\mathrm{TOT}}\{\phi_w,\,\phi_\theta\}^\top = q\beta\,\mathbf{F}_a$ gives
\begin{align}
\phi_w &= \frac{q\beta}{\det\mathbf{K}_{\mathrm{TOT}}}
\left(K_{TOT,22}F_{a1} - K_{TOT,12}F_{a2}\right), \\
\phi_\theta &= \frac{q\beta}{\det\mathbf{K}_{\mathrm{TOT}}}
\left(K_{TOT,11}F_{a2} - K_{TOT,21}F_{a1}\right),
\end{align}
hence
\begin{equation}
    \boxed{ z(x,y) = \frac{q\beta}{\det\mathbf{K}_{\mathrm{TOT}}}\frac{y}{b}
\left[\frac{y}{b}\left(K_{TOT,22}F_{a1}-K_{TOT,12}F_{a2}\right)
- x\left(K_{TOT,11}F_{a2}-K_{TOT,21}F_{a1}\right)\right].}
\end{equation}

\paragraph{Configurations compared.}
\begin{itemize}
  \item \textbf{With strut:} full structural model;
  \item \textbf{Without strut:} strut axial stiffness set to $\mathrm{EA}=0$.
\end{itemize}

\noindent To achieve the comparison we define:
\begin{equation}
\delta_z=100\,\frac{|z_{\mathrm{strut}}-z_{\mathrm{ns}}|}{|z_{\mathrm{ns}}|}.
\end{equation}
Which is the relative deviation of the strut configuration with respect to the no-strut reference.

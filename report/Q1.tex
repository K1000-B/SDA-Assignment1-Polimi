
\renewcommand{\parttitle}{Divergence Dynamic Pressure $q_D$ - Derivation}
% ============================================================
\section*{Problem Setup}
% ============================================================

The wing has two DOFs --- bending $w(y)$ and torsion $\theta(y)$ --- with a strut at $y = y_B$,
aerodynamic center offset $e$ from the elastic axis, and a tip aileron of span $b_a$ and deflection $\beta$.
The Ritz--Galerkin method is applied to perform a static aeroelastic analysis.

% ============================================================
\section*{\parttitle}
% ============================================================

\subsection*{Shape Functions}

Root clamp imposes $w(0)=w'(0)=\theta(0)=0$. The Ritz ansatz
$w = N_w(y)\,\phi_w$, $\theta = N_\theta(y)\,\phi_\theta$
requires $N_w$ quadratic and $N_\theta$ linear. Applying BCs and normalizing to unity at the tip:

\begin{equation}
\boxed{N_w(y) = \left(\frac{y}{b}\right)^2, \qquad N_\theta(y) = \frac{y}{b}}
\end{equation}

% ============================================================
\subsection*{Principle of Virtual Work}
% ============================================================

In order to compute $q_D$, we need to find the equations of motion. To do so, we use the PVW : We need to compute the internal and external virtual work.

\subsubsection*{Internal Virtual Work}

Substituting $N_w'' = 2/b^2$ and $N_\theta' = 1/b$:

\begin{align}
\delta W_i = \int_0^b \delta w_{,yy}\,EJ\,w_{,yy}\,dy + \int_0^b \delta\theta_{,y}\,GJ\,\theta_{,y}\,dy = \delta\phi_w\,\frac{4EJ}{b^3}\,\phi_w + \delta\phi_\theta\,\frac{GJ}{b}\,\phi_\theta
\end{align}

\begin{equation}
\boxed{
\delta W_i = \delta\boldsymbol{\phi}^T
\begin{bmatrix} K_w & 0 \\ 0 & K_\theta \end{bmatrix}
\boldsymbol{\phi}, \qquad
K_w = \frac{4EJ}{b^3}, \quad K_\theta = \frac{GJ}{b}
}
\end{equation}

% ============================================================
\subsubsection*{External Virtual Work}
% ============================================================

\[
\delta W_e = \delta W_{\text{AERO}}^{\text{wing}} + \delta W_{\text{AERO}}^{\text{aileron}} + \delta W_{\mathrm{STRUT}} + \delta W_{\text{INERTIA}} + \delta W_{\text{GRAVITY}}
\]

Since we are interested in the divergence dynamic pressure $q_D$, we can neglect inertia and gravity (static case, no external forces). Thus:
\[
\boxed{\delta W_e = \delta W_{\text{A}}^{\text{w}} + \delta W_{\text{A}}^{\text{a}} + \delta W_{\mathrm{STR}}}
\]


\paragraph{(1) Wing aerodynamic virtual work $\delta W_A^w$}

The kinematically admissible AC displacement is $h_{AC} = w + e\,\theta$, and strip lift reads
$L'_w = q\,c\,C_{L/\alpha}\,\theta$ (perturbation, $\alpha_\mathrm{eff}=\theta$, $C_{L_0}=0$). Thus:

\begin{align}
\delta W_A^w &= \int_0^b \delta h_{AC}\,L'_w\,dy
= \int_0^b \delta\phi_w \left(\frac{y}{b}\right)^2 q\,c\,C_{L/\alpha}\,\frac{y}{b}\,\phi_\theta\,dy
+ \int_0^b \delta\phi_\theta\,\frac{y}{b}\,e\,q\,c\,C_{L/\alpha}\,\frac{y}{b}\,\phi_\theta\,dy
\end{align}

\begin{equation}
\boxed{
\delta W_A^w = \delta\boldsymbol{\phi}^T\,q\,\mathbf{K}_A\,\boldsymbol{\phi}
}, \qquad
\mathbf{K}_A =
\begin{bmatrix}
0 & \dfrac{cb}{4}C_{L/\alpha} \\[6pt]
0 & \dfrac{ecb}{3}C_{L/\alpha}
\end{bmatrix}
\end{equation}

$\mathbf{K}_A$ is the aerodynamic stiffness matrix.


\paragraph{(2) Aileron aerodynamic virtual work $\delta W_A^a$}

Not needed for $q_D$ (homogeneous problem) but derived for later use. The aileron spans $[b-b_a,\,b]$
with deflection $\beta$; perturbation gives $C_{m_{AC}} = C_{m/\beta}\beta$. The virtual work splits into
a lift contribution and a pitching moment contribution:

\begin{align}
\delta W_A^a &= \int_{b-b_a}^{b} \delta(w+e\theta)\,q\,c\,C_{L/\beta}\,\beta\,dy
+ \int_{b-b_a}^{b} \delta\theta\,q\,c^2\,C_{m/\beta}\,\beta\,dy
\end{align}

\begin{equation}
    \label{eq:Fa}
\boxed{
\delta W_A^a = \delta\boldsymbol{\phi}^T\,q\beta\,\mathbf{F}_a
}, \qquad
\mathbf{F}_a = \frac{c}{b}
\begin{Bmatrix}
C_{L/\beta}\cdot\dfrac{b^3-(b-b_a)^3}{3b} \\[8pt]
\left(e\,C_{L/\beta}+c\,C_{m/\beta}\right)\dfrac{b^2-(b-b_a)^2}{2}
\end{Bmatrix} 
=\begin{Bmatrix}
F_{a1} \\
F_{a2}
\end{Bmatrix}
\end{equation}

The strut axial force and its $z$-component follow from geometry ($\ell = z_A/\sin\gamma$,
$u = h_B\sin\gamma$):

\begin{equation}
\boxed{F_{Sz} = -\underbrace{\frac{EA}{z_A}\sin^3\gamma}_{K_\mathrm{STR}} h_B}
\end{equation}

The displacement at attachment point B, using $\eta_B = z_A/(b\tan\gamma)$:

\[
h_B = w(y_B) - x_B\,\theta(y_B) = \eta_B\bigl[\eta_B\,\phi_w - x_B\,\phi_\theta\bigr]
\]

Substituting into $\delta W_\mathrm{STR} = \delta h_B\,(-K_\mathrm{STR})\,h_B$:

\begin{equation}
\boxed{
\delta W_\mathrm{STR} = -\delta\boldsymbol{\phi}^T\,K_\mathrm{STR}\,\eta_B^2
\begin{bmatrix} \eta_B^2 & -\eta_B x_B \\ -\eta_B x_B & x_B^2 \end{bmatrix}
\boldsymbol{\phi}
}
\end{equation}


\subsection*{Application of PVW: $\delta W_i = \delta W_e$}

\begin{equation}
\delta\begin{Bmatrix}\phi_w \\ \phi_\theta\end{Bmatrix}^T
\begin{bmatrix} K_w & 0 \\ 0 & K_\theta \end{bmatrix}
\begin{Bmatrix}\phi_w \\ \phi_\theta\end{Bmatrix}
=
\delta\begin{Bmatrix}\phi_w \\ \phi_\theta\end{Bmatrix}^T 
\left(
q\mathbf{K}_A \begin{Bmatrix}\phi_w \\ \phi_\theta\end{Bmatrix}
+ q\,\beta\,\mathbf{F}_{\mathrm{a}}
- K_{\mathrm{STR}}\,\eta_B^2\begin{bmatrix}\eta_B^2 & -\eta_B x_B \\ -\eta_B x_B & x_B^2\end{bmatrix} \begin{Bmatrix}\phi_w \\ \phi_\theta\end{Bmatrix}
\right)
\end{equation}

Assembling all contributions and invoking the arbitrariness of $\delta\boldsymbol{\phi}$
(unforced case $\beta = 0$ for $q_D$ computation):

\begin{equation}
\boxed{
\left(
\begin{bmatrix}
K_w + K_{\mathrm{STR}}\,\eta_B^4 & -K_{\mathrm{STR}}\,\eta_B^3\,x_B \\
-K_{\mathrm{STR}}\,\eta_B^3\,x_B & K_\theta + K_{\mathrm{STR}}\,\eta_B^2\,x_B^2
\end{bmatrix}
- q\,\mathbf{K}_A
\right)
\begin{Bmatrix}\phi_w \\ \phi_\theta\end{Bmatrix}
=
\begin{Bmatrix}0\\0\end{Bmatrix}
}
\end{equation}

This is $\left(\mathbf{K}_S - q_D\,\mathbf{K}_A\right)\boldsymbol{\phi} = \mathbf{0}$, with:

\begin{equation}
\mathbf{K}_S =
\begin{bmatrix}
K_{S_1} & K_{S_{12}} \\
K_{S_{12}} & K_{S_2}
\end{bmatrix},
\qquad
\mathbf{K}_A =
\begin{bmatrix}
0 & \mathbf{K_{A_1}}=\dfrac{cb}{4}\,C_{L/\alpha} \\[6pt]
0 & \mathbf{K_{A_2}}=\dfrac{ecb}{3}\,C_{L/\alpha}
\end{bmatrix}
\end{equation}

\begin{align}
K_{S_1} = \frac{4EJ}{b^3} + \frac{EA\,z_A^3}{b^4}\,\frac{\cos^4\gamma}{\sin\gamma}, \qquad K_{S_{12}} = -\frac{EA\,z_A^2\,\cos^3\gamma\,x_B}{b^3}, \qquad K_{S_2} = \frac{GJ}{b} + \frac{EA\,\sin\gamma\,\cos^2\gamma\,z_A\,x_B^2}{b^2}
\end{align}

For divergence, the condition is: $\det\!\left(\mathbf{K}_S - q_D\,\mathbf{K}_A\right) = 0$. Expanding the determinant:
\begin{equation}
    \label{eq:det_KTOT}
\boxed{\det\!\left(\mathbf{K}_S - q_D\,\mathbf{K}_A\right) = K_{S_1}K_{S_2} - q_D\,K_{S_1}\,K_{A_2} - K_{S_{12}}^2 + q_D\,K_{S_{12}}\,K_{A_{1}}} = 0 \quad \Rightarrow \quad
\boxed{\textcolor{red!90!black}{q_D = \frac{K_{S_1}\,K_{S_2} - (K_{S_{12}})^2}{K_{S_1}\,K_{A_2} - K_{S_{12}}\,K_{A_{1}}}}}
\end{equation}
